 23  * @param {cpr.controls.layouts.FormDivision} div
 24  */
 25 function copyDiv(div) {
 26 	var result = new cpr.controls.layouts.FormDivision(div.lengthExpression);
 27 	result.autoSizing = div.autoSizing;
 28 	result.hidden = div.hidden;
 29 	result.shades = div.shades;
 30 	result.customShadeColor = div.customShadeColor;
 31 	result.minLength = div.minLength;
 32 	return result;
 33 }

 73 /**
 74  * 초기 상태 백업.
 75  */
 76 RForm.prototype._backup = function() {
 77 	// 원래 컨스트레인트들.
 78 	this._originalConstraints = new cpr.utils.ObjectMap();
 79 
 80 	var children = this._container.getChildren();
 81 	for (var idx = 0; idx < children.length; idx++) {
 82 		var eachChild = children[idx];
 83 		var constraint = this._container.getConstraint(eachChild);
 84 		this._originalConstraints.put(eachChild, constraint);
 85 	}
 86 
 87 	this._originalLayout = this._container.getLayout();
 88 };

 90 RForm.prototype.start = function() {
 91 	if (this._started) {
 92 		return;
 93 	}
 94 	this._backup();
 95 	this._appInstance = this._container.getAppInstance();
 96 	this._appInstance.addEventListener("screen-change", this._onScreenChange);
 97 	this._container.addEventListenerOnce("dispose", this.stop.bind(this));
 98 	this._started = true;
 99 };
100 
101 RForm.prototype.stop = function() {
102 	if (!this._started) {
103 		return;
104 	}
105 	if (this._appInstance) {
106 		this._appInstance.removeEventListener("screen-change", this._onScreenChange);
107 		this._appInstance = null;
108 	}
109 
110 	this._started = false;
111 };

113 RForm.prototype._restore = function() {
114 	this._container.setLayout(this._originalLayout);
115 	/** @type cpr.controls.VisibleUIControl[] */
116 	var children = this._container.getChildren();
117 	for (var idx = 0; idx < children.length; idx++) {
118 		var eachChild = children[idx];
119 		eachChild.visible = true;
120 		this._container.replaceConstraint(eachChild, this._originalConstraints.get(eachChild));
121 	}
122 	this._clearCollapseClasses();
123 }

127  * @param {cpr.events.CScreenChangeEvent} e
128  */
129 RForm.prototype._onScreenChange = function(e) {
130 	var colSettings = this._columnSettings[e.screen.name];
131 	if (colSettings == null || colSettings <= 0) {
132 		this._restore();
133 	} else {
134 		this._transform(colSettings);
135 	}

138 /**
139  * @return {cpr.controls.VisibleUIControl[]}
140  */
141 RForm.prototype._getSortedChildren = function() {
142 	var children = this._container.getChildren().slice(0);
143 	var me = this;
144 	var useVerticalFirstSort = this._container.userAttr(ATTR_COLLAPSE_DIRECTION) == "vertical";
145 
146 	/** @type String */
147 	var explictIndexAttribute = null;
148 	switch (this._container.getAppInstance().targetScreen.name) {
149 		case "mobile":
150 			{
151 				explictIndexAttribute = ATTR_MOBILE_INDEX;
152 				break;
153 			}
154 
155 		case "tablet":
156 			{
157 				explictIndexAttribute = ATTR_TABLET_INDEX;
158 				break;
159 			}
160 	}
161 
162 	return children.sort(function( /* cpr.controls.UIControl*/ a, /* cpr.controls.UIControl*/ b) {
163 
164 		var indexA = parseInt(a.userAttr(explictIndexAttribute) || "9999");
165 		var indexB = parseInt(b.userAttr(explictIndexAttribute) || "9999");
166 		var explictIndexGap = indexA - indexB;
167 		if (explictIndexGap !== 0) {
168 			return explictIndexGap;
169 		}
170 
171 		/** @type cpr.controls.layouts.FormConstraint */
172 		var constA = me._originalConstraints.get(a);
173 
174 		/** @type cpr.controls.layouts.FormConstraint */
175 		var constB = me._originalConstraints.get(b);
176 		var rowDiff = constA.rowIndex - constB.rowIndex;
177 		var colDiff = constA.colIndex - constB.colIndex;
178 
179 		if (useVerticalFirstSort) {
180 			return colDiff !== 0 ? colDiff : rowDiff;
181 		} else {
182 			return rowDiff !== 0 ? rowDiff : colDiff;
183 		}
184 	});
185 };

187 /**
188  * 
189  * @param {Number} colCount
190  */
191 RForm.prototype._transform = function(colCount) {
192 	var layout = new cpr.controls.layouts.FormLayout();
193 	this._container.setLayout(layout);
194 
195 	layout.horizontalMargin = this._originalLayout.horizontalMargin;
196 	layout.verticalMargin = this._originalLayout.verticalMargin;
197 	layout.horizontalSpacing = this._originalLayout.horizontalSpacing;
198 	layout.verticalSpacing = this._originalLayout.verticalSpacing;
199 	layout.horizontalSeparatorWidth = this._originalLayout.horizontalSeparatorWidth;
200 	layout.horizontalSeparatorClass = this._originalLayout.horizontalSeparatorClass;
201 	layout.verticalSeparatorWidth = this._originalLayout.verticalSeparatorWidth;
202 	layout.verticalSeparatorClass = this._originalLayout.verticalSeparatorClass;
203 
204 	switch (this._appInstance.targetScreen.name) {
205 		case "mobile":
206 			if (this._container.userAttr(ATTR_MOBILE_H_MARGIN)) {
207 				layout.horizontalMargin = this._container.userAttr(ATTR_MOBILE_H_MARGIN);
208 			}
209 			if (this._container.userAttr(ATTR_MOBILE_V_MARGIN)) {
210 				layout.verticalMargin = this._container.userAttr(ATTR_MOBILE_V_MARGIN);
211 			}
212 			break;
213 
214 		case "tablet":
215 			if (this._container.userAttr(ATTR_TABLET_H_MARGIN)) {
216 				layout.horizontalMargin = this._container.userAttr(ATTR_TABLET_H_MARGIN);
217 			}
218 			if (this._container.userAttr(ATTR_TABLET_V_MARGIN)) {
219 				layout.verticalMargin = this._container.userAttr(ATTR_TABLET_V_MARGIN);
220 			}
221 			break;
222 
223 	}
224 
225 	var columnDivisions = this._originalLayout.getColumnDivisions();
226 	var colSettings = columnDivisions.slice(0, colCount);
227 
228 	// 모바일이나 태블릿이 컬럼 개수가 더 많은 경우 처리.
229 	while (colSettings.length < colCount) {
230 		colSettings.push(copyDiv(colSettings[colSettings.length - 1]));
231 	}
232 
233 	var rowDiv = new cpr.controls.layouts.FormDivision("1fr");
234 	var vnRowMinLength = 10;
235 	
236 	var rowDivisions = this._originalLayout.getRowDivisions();
237 	colSettings.forEach(function( /* cpr.controls.layouts.FormDivision */ each, idx) {
238 		if(rowDivisions[idx]){
239 			var vsLengthExp  = rowDivisions[idx].lengthExpression;
240 			vsLengthExp = vsLengthExp.replace("px", "");
241 			var vnLengthExp = ValueUtil.fixNumber(vsLengthExp);
242 			if(vnRowMinLength < vnLengthExp){
243 					vnRowMinLength = vnLengthExp;
244 			}	
245 		}
246 		
247 	});
248 	if(vnRowMinLength > 10){
249 		rowDiv.minLength =  vnRowMinLength;	
250 	}
251 	
252    		
253 
254 	// 세로 컬랩스 처리.
255 	if (this._container.userAttr(ATTR_COLLAPSE_DIRECTION) == "vertical") {
256 		var rowDivisions = this._originalLayout.getRowDivisions();
257 		colSettings.forEach(function( /* cpr.controls.layouts.FormDivision */ each, idx) {
258 			each.shades = rowDivisions[idx].shades;
259 			each.customShadeColor = rowDivisions[idx].customShadeColor;
260 		});
261 		rowDiv.shades = false;
262 	}
263 
264 	var lastColDiv = colSettings[colSettings.length - 1];
265 	lastColDiv.lengthExpression = "1fr";
266 	lastColDiv.shades = false;
267 	layout.setColumnDivisions(colSettings);
268 
269 	var numberOfColumns = colSettings.length;
270 	var sortedChildren = this._getSortedChildren();
271 
272 	var visibleChildren = sortedChildren.filter(function( /* cpr.controls.VisibleUIControl */ each) {
273 		return each.userAttr("hide-on-mobile") != "true";
274 	});
275 
276 	var hiddenChildren = sortedChildren.filter(function( /* cpr.controls.VisibleUIControl */ each) {
277 		return each.userAttr("hide-on-mobile") == "true";
278 	}).forEach(function( /* cpr.controls.VisibleUIControl */ each) {
279 		each.visible = false;
280 	});
281 
282 	var rowIndex = 0;
283 	var columnIndex = 0;
284 	var rowNeedsAutoHeight = false;
285 
286 	/** @type cpr.controls.layouts.FormDivision[] */
287 	var rows = [];
288 
289 	for (var idx = 0; idx < visibleChildren.length; idx++) {
290 		if (rows[rowIndex] == null) {
291 			rows.push(copyDiv(rowDiv));
292 		}
293 		/**
294 		 * 현재 행에 남은 컬럼 수.
295 		 */
296 		var leftColumnsInCurrentLine = numberOfColumns - columnIndex;
297 		var each = visibleChildren[idx];
298 		each.visible = true;
299 
300 		/**
301 		 * 원래 컨스트레인트.
302 		 * @type cpr.controls.layouts.FormConstraint
303 		 */
304 		var originalConstraint = this._originalConstraints.get(each);
305 		var originalColSpan = originalConstraint.colSpan || 1;
306 
307 		var colSpan = Math.min(originalColSpan, leftColumnsInCurrentLine);
308 
309 		this._container.replaceConstraint(each, {
310 			rowIndex: rowIndex,
311 			colIndex: columnIndex,
312 			colSpan: colSpan
313 		});
314 
315 		if (each.userAttr(ATTR_NEEDS_AUTO_HEIGHT) == "true") {
316 			rows[rows.length - 1].autoSizing = true;
317 			rows[rows.length - 1].lengthExpression = "25px";
318 		}
319 
320 		if (each.userAttr(ATTR_MOBILE_MIN_HEIGHT)) {
321 			var eachMinLength = parseInt(each.userAttr(ATTR_MOBILE_MIN_HEIGHT));
322 			var knownRow = rows[rows.length - 1];
323 			if (knownRow.lengthExpression.match(".*fr$")) {
324 				knownRow.minLength = Math.max(knownRow.minLength, eachMinLength);
325 			} else {
326 				knownRow.lengthExpression = Math.max(parseInt(knownRow.lengthExpression), eachMinLength) + "px";
327 			}
328 		}
329 
330 		columnIndex = columnIndex + colSpan;
331 
332 		if (columnIndex >= numberOfColumns) {
333 			rowIndex++;
334 			columnIndex = 0;
335 		}
336 	}
337 
338 	layout.setRowDivisions(rows);
339 
340 	this._clearCollapseClasses();
341 	this._container.style.addClass("collapsed");
342 	this._container.style.addClass("collapsed-" + colCount);
343 };

345 /**
346  * 
347  * @param {Number} colCount
348  */
349 RForm.prototype._clearCollapseClasses = function() {
350 	var classes = this._container.style.getClasses();
351 	var collapseClasses = classes.filter(function( /* String */ each) {
352 		return each.indexOf("collapsed") === 0;
353 	});
354 	collapseClasses.forEach((function( /* String */ each) {
355 		this._container.style.removeClass(each);
356 	}).bind(this));
357 }

373 RGrid.prototype._handleDispose = function(e) {
374 	this._app.removeEventListener("screen-change", this._onScreenChange);
375 };

377 /**
378  * @param {cpr.events.CScreenChangeEvent} e
379  */
380 RGrid.prototype._onScreenChange = function(e) {
381 	switch (e.screen.name) {
382 		case "mobile":
383 			{
384 				this.collapse();
385 				break;
386 			}
387 
388 		default:
389 			{
390 				this.restore();
391 				break;
392 			}
393 	}

396 RGrid.prototype.collapse = function() {
397 	var indicies = this._grid.userAttr(ATTR_HIDE_CELL_INDICIES).split(/[\s,]+/g).map(function( /* String */ each) {
398 		return parseInt(each);
399 	});
400 	var grid = this._grid;
401 	indicies.forEach(function( /* Number */ each) {
402 		if(grid.header.getColumn(each)){
403 			grid.header.getColumn(each).visible = false;	
404 		}
405 	});
406 };

414 		if(grid.header.getColumn(each)){
415 			grid.header.getColumn(each).visible = true;	
416 		}
417 	});
