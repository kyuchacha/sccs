 11 /*
 12  * 본 모듈은 MDI폴더에서 열려있는 텝 아이템을 가져옵니다.
 13  * 불러온 탭아이텝 리스트를 통해 특정, 또는 모든 아이템을 닫을 수 있습니다.
 14  * 일부를 닫을 경우에는, 현재 탭폴더에 열려있는 탭아이템을 체크박스 그룹으로 확인해 닫을 수 있습니다.
 15  * [닫기] 또는 빈 공간을 누르면 체크박스그룹은 사라집니다.
 16  * [분할] 을 누르면 선택한 아이템을 분할해서 볼 수 있습니다. 단, [분할]하기 위해서 DivideMdi.module.js 모듈과 함께 사용해야합니다.
 17  * DivideMdi.module.js 는 [게시판 > eXbuilder6자료 > MDI폴더 분할 공통모듈] 에서 다운로드 받을 수 있습니다.
 18  * 
 19  * [사용가능한 메서드]
 20  * 1. getTabList : 열려있는 텝 아이템을 호출한 뒤, [분할] or [닫기]를 할 수 있습니다.
 21  * 2. closeTabAll : 열려있는 모든 텝아이템을 닫습니다.
 22  */

364 /**
365  * 열려있는 탭아이템 리스트를 추가합니다. (체크박스그룹)
366  * @param {cpr.core.AppInstance} poApp
367  * @param {cpr.controls.TabFolder | cpr.controls.MDIFolder} pcMDIFolder
368  * @param {cpr.controls.Container} pcParent 리스트를 추가한 부모그룹
369  */
370 function _insertOpenItems (poApp, pcMDIFolder, pcParent, comDiv) {
371 	
372 	var vcDataSet = poApp.lookup("dsOpenItems");
373 	if(vcDataSet == undefined || vcDataSet == null) { // 생성한 데이터셋은 최초한번만 register 하도록 수정
374 		vcDataSet = new cpr.data.DataSet("dsOpenItems");
375 		vcDataSet.parseData({
376 			columns : [
377 				{dataType: "string", name: "label"},
378 				{dataType: "string", name: "value"},
379 				{dataType: "string", name: "appId"},
380 			]
381 		});
382 		poApp.register(vcDataSet);
383 	}
384 	
385 	if(pcMDIFolder) {
386 		var voTabItems = pcMDIFolder.getTabItems();
387 		maCascading = [];
388 		
389 		// 계단형에서 다른형태로 분할 시 삭제된 아이템을 누락할 수 있도록 수정
390 		var vaDialogNms = poApp.dialogManager.getDialogNames().filter(function(each){
391 			return each.indexOf("ea") != -1;
392 		});
393 		if(vaDialogNms.length > 0) {
394 			// 계단형일 경우
395 			var voLoadApps = cpr.core.Platform.INSTANCE.getAllLoadedApps();
396 			vaDialogNms.forEach(function(each, index){
397 				maCascading.push({
398 					label :  poApp.dialogManager.getDialogByName(each).headerTitle,
399 					value : each,
400 					appId : each.slice(2, each.lastIndexOf("_"))
401 				});
402 			});
403 		}
404 		
405 		if(maCascading.length > 0) {
406 			maCascading.forEach(function(each){
407 				voTabItems.push(each)
408 			});
409 		}
410 		
411 		vcDataSet.clear();
412 		for(var idx = 0; idx < voTabItems.length; idx++){
413 			
414 			if(voTabItems[idx].text != "tempItem") {
415 				var vsLabel = "";
416 				var vsValue = "";
417 				var vsAppId = "";
418 				
419 				if(voTabItems[idx] instanceof cpr.controls.TabItem) {
420 					vsLabel = voTabItems[idx].text;
421 					vsValue = voTabItems[idx].itemIndex;
422 				} else if( voTabItems[idx] instanceof cpr.controls.Item){
423 					vsLabel = voTabItems[idx].row.getValue("label");
424 					vsValue = voTabItems[idx].row.getValue("value");
425 					vsAppId = voTabItems[idx].row.getValue("appId");
426 				} else {
427 					vsLabel = voTabItems[idx].label;
428 					vsValue = voTabItems[idx].value;
429 					vsAppId = voTabItems[idx].appId;
430 				}
431 				
432 				if(voTabItems[idx] instanceof cpr.controls.TabItem && pcMDIFolder.type == "mdifolder") {
433 					if(voTabItems[idx].content && voTabItems[idx].content.app) {
434 						vsAppId = voTabItems[idx].content.app.id;						
435 					} else {
436 						// 분할 된 상태에서 새로 추가한 메뉴에 대한 appId 설정
437 						/** @type cpr.controls.Container */
438 						var vcGrpFloat = comDiv.getDivGroup();
439 						if(vcGrpFloat) {
440 							var vcEmb = vcGrpFloat.getChildren().filter(function(each){
441 								return each.uuid == voTabItems[idx].name;
442 							})[0];
443 							vsAppId = vcEmb.app.id;		
444 						} else {
445 							var vsTempId = voTabItems[idx].content.id;
446 							vsAppId = vsTempId.slice(2, vsTempId.lastIndexOf("_"));
447 						}
448 					}
449 				}
450 				
451 				vcDataSet.addRowData({
452 					label : vsLabel,
453 					value : vsValue,
454 					appId : vsAppId
455 				})
456 			}
457 		}
458 	}
459 	
460 	// 체크박스를 추가할 버티컬레이아웃
461 	var voVerticalLayout = pcParent.getAllRecursiveChildren().filter(function(each) {
462 		if(each.id == "grpList") {
463 			return each;
464 		}
465 	})[0];
466 	
467 	
468 	// 체크박스 추가(전체선택)
469 	var vcCheckBox = new cpr.controls.CheckBox();
470 	vcCheckBox.text = "전체선택";
471 	vcCheckBox.style.addClass(msCheckBoxCls);
472 	vcCheckBox.addEventListener("value-change", _onCbxValueChange);
473 	voVerticalLayout.addChild(vcCheckBox, {
474 		"autoSize": "height",
475 		"width": "200px",
476 		"height": "30px"
477 	});
478 	
479 	if(mcCmbLayout) {
480 		var vnItemCnt = pcMDIFolder.getTabItems().length + maCascading.length;
481 		if(mcCmbLayout.value == "default" || mcCmbLayout.value == "division") {
482 			vcCheckBox.enabled = false;
483 		}
484 	}
485 	
486 	// 체크박스 그룹 추가 (현재 열려있는 탭아이템 목록)
487 	// TODO 체크박스그룹의 속성 및 스타일을 설정하십시오.
488 	var vcOpenItemCbg = new cpr.controls.CheckBoxGroup();
489 	vcOpenItemCbg.colCount = Math.ceil(vcDataSet.getRowCount()/4);
490 	vcOpenItemCbg.addEventListener("selection-change", _onOpenItemSelectionChange);
491 	vcOpenItemCbg.setItemSet(vcDataSet, {
492 		"label": "label",
493 		"value": "value"
494 	});
495 	voVerticalLayout.addChild(vcOpenItemCbg, {
496 		"autoSize": "height",
497 		"width": "200px",
498 		"height": "30px"
499 	})
500 	
501 	// 고정탭아이템 있을 때 해당 체크박스 아이템 disable 처리
502 	if(mbFixed) {
503 		var vsExp = "";
504 		vcOpenItemCbg.enabledItemExp = vsExp;
505 	}
506 }

581 /**
582  * 선택한 탭아이템 닫기
583  * @param {cpr.controls.UIControl} pcControl
584  * @param {cpr.controls.TabFolder | cpr.controls.MDIFolder} pcMDIFolder
585  */
586 function _closeGroup (pcControl, pcMDIFolder, comDiv) {
587 
588 	var vcParent = pcControl.getParent();
589 
590 	if(vcParent) {
591 		/** @type cpr.controls.CheckBox */
592 		var vcCheckAll = null;
593 		/** @type cpr.controls.CheckBoxGroup */
594 		var vcCheckList = null;
595 		
596 		vcParent.getAllRecursiveChildren().forEach(function(each){
597 			if(each.type == "checkboxgroup") {
598 				vcCheckList = each;
599 			} else if( each.type == "checkbox") {
600 				vcCheckAll = each;
601 			}
602 		});		
603 		
604 		if(vcCheckAll && vcCheckList) {
605 			var voSelection = vcCheckList.getSelection();
606 			voSelection.forEach(function(each){
607 				var vsValue = each.row.getValue("value");
608 				var vsAppId = each.row.getValue("appId");
609 				var vcClosableItem = null;
610 				
611 				if(pcMDIFolder.type == "tabfolder") {
612 					vcClosableItem = pcMDIFolder.getTabItemByID(vsValue);
613 				} else {
614 					vcClosableItem = pcMDIFolder.findItemWithAppID(vsAppId);
615 				}
616 				
617 				var voRootAppIns = pcControl.getAppInstance().getRootAppInstance();
618 				if(voRootAppIns.lookup("grpFloat")) {
619 					comDiv.closeMenu(mcCmbLayout, "select", vcClosableItem);
620 				}
621 				
622 				pcMDIFolder.close(vcClosableItem);
623 			});
624 			
625 			// 그룹 삭제
626 			_removeFloating(pcControl.getParent(), comDiv);
627 		}
628 	}
629 }

649 /**
650  * 체크박스 그룹의 selection-change 이벤트리스너
651  * @param {cpr.events.CUIEvent} e
652  */
653 function _onOpenItemSelectionChange (e) {
654 	
655 	/** @type cpr.controls.CheckBoxGroup */
656 	var control = e.control
657 	var vcParent = control.getParent();
658 
659 	/** @type cpr.controls.CheckBox */
660 	var vcCbxAll = vcParent.getChildren().filter(function(each) {
661 		if(each.type == "checkbox"){
662 			return each;
663 		}
664 	})[0];
665 	
666 	/* 체크박스그룹을 모두 체크하면 모두 선택도 체크 */
667 	if (control.getSelection().length == control.getItemCount()){
668 		vcCbxAll.putValue("true");
669 	} else {
670 		vcCbxAll.putValue("");
671 	}
672 	
673 	if(mcCmbLayout) {
674 		if(mcCmbLayout.value == "default") {
675 			_enabledCbg(control, 1);
676 		}
677 		if(mcCmbLayout.value == "division") {
678 			_enabledCbg(control, 4);
679 		}
680 	}
681 }

683 /**
684  * 체크박스그룹 아이템 enabled 설정
685  * @param {cpr.controls.CheckBoxGroup} pcCbg
686  * @param {Number} vnSelection 최대 선택 개수
687  */
688 function _enabledCbg (pcCbg, vnSelection) {
689 	var voSelection = pcCbg.getSelection();
690 	var voEnable = [];
691 	voSelection.forEach(function(each){
692 		voEnable.push("value == '" + each.value + "'");
693 	});
694 
695 	if(voSelection.length == vnSelection) {
696 		var vsExp = voEnable.join(" || ");
697 		pcCbg.enabledItemExp = vsExp;
698 	} else {
699 		pcCbg.enabledItemExp = "";
700 	}
701 }
