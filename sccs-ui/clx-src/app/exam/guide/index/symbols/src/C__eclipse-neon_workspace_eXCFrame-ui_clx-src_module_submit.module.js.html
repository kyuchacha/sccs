 28  */
 29 function _requestEncoder(submission, reqData) {
 30 	var _app = submission.getAppInstance();
 31 	var reqDataObj = reqData["data"];
 32 	var paramObj = reqData["param"];
 33 	
 34 	var voJsonType = {};
 35 	for (var key in reqDataObj) {
 36 		voJsonType[key] = reqDataObj[key];
 37 	}
 38 	
 39 	return {"content" : voJsonType};
 40 }

 43 /**
 44  * @private
 45  * @param {cpr.protocols.Submission} submission
 46  * @param {Object} reqData
 47  */
 48 function _responseDecoder(submission, resData) {
 49 	var _app = submission.getAppInstance();
 50 	var resDataObj = JSON.parse(resData);
 51 	
 52 	var voProtocolJson = {};
 53 	
 54 	for (var key in resDataObj) {
 55 		for (var subKey in resDataObj[key]) {
 56 			voProtocolJson[subKey] = resDataObj[key][subKey];
 57 		}
 58 	}
 59 	
 60 	return {contentType : "application/json", content : voProtocolJson};
 61 }

 97 function uf_send(e){
 98 	  
 99 	var thisCtl = e.control ; 
100 	var vsParamListObj = new cpr.utils.ObjectMap();
101 /**
102  * @type cpr.protocols.Submission
103  */
104 	var vSub =  e.targetControl ;
105 
106 	if (vSub == null || vSub.type == null) {
107 		return;
108 	}
109 	
110 	//apply url petten ( true , false ) 
111 	//default value  true 
112 	var vsUAttParm = vSub.userAttr("uri_parameter_gen") ;
113 
114 	if( "true" != vsUAttParm ) return ;
115 	
116 	var vsParamList    = vSub.getParameterNames(); // send parameter ( remamber basic config )
117 	var vnSendParCnt   = vSub.getRequestDataCount(); 
118 	var vsSendParList  = vSub.getRequestObject(); // send data ( dataSet or dataMap )
119 	
120 	vsParamList.forEach( function ( each1 ) {
121 		vsParamListObj.put(each1, vSub.getParameters(each1));
122 	});
123 	 
124 	 
125 	var isDataMap = true ;
126 	var isDataSet = true ;
127 	var vsUAttParmMap = vSub.userAttr("uri_parameter_gen_dataMap") ;
128 	var vsUAttParmSet = vSub.userAttr("uri_parameter_gen_dataSet") ;
129 	if( vsUAttParmMap == "false" ) isDataMap = false ; 
130 	if( vsUAttParmSet == "false" ) isDataSet = false ; 
131 	 
132 	 
133 	for( var i = 0 ; i < vnSendParCnt ; i++){
134 		var vds1  = vSub.getRequestData(i); 
135 	    
136 		  //dataMap 이 아니면 이를 적용하지 않는다.  
137 		  if( ! ( vds1 instanceof cpr.data.DataMap )   ){
138 		  	  if( ! isDataMap )
139 		  	      continue;
140 		  }
141 		  
142 		  //dataSet 이 아니면 이를 적용하지 않는다. 
143 		  //TODO : dataSet의 기능이 필요할 경우 이를 활용한다.   
144 		  if( ! ( vds1 instanceof cpr.data.DataSet )   ){
145 		  	  if( ! isDataSet )
146 		  	      continue;
147 		  }
148 		  
149 		  var vdsData = vds1.data ;
150 		  
151 		  var vds1ColNm = vdsData.getColumnNames();
152 		  
153 		  //column을 param으로 이동 시킨다. 
154 		  vds1ColNm.forEach( function(eachPaNm){
155 		  	   var vsParValue = vdsData.getString(eachPaNm);
156 		  	   vSub.setParameters(eachPaNm, vsParValue); 
157 		  }); 
158 	}
159  
160     vSub.addEventListenerOnce("submit-done", function(){
161 		    /**
162 		      * 파라미터를 삭제하고 복원한다. 
163 		      */
164 		    vSub.removeAllParameters();
165 		    vsParamListObj.forEach(function(eachMap){  
166 		       vSub.setParameters(eachMap, vsParamListObj.get(eachMap));  
167 		    }); 
168     });
169 }
