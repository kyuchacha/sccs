 33 /**
 34  * 화면에 LoadMask 출력
 35  * 비동기 서브미션 호출시 화면에 로딩 이미지 출력
 36  * @param {cpr.core.AppInstance} app 앱인스턴스
 37  * @param {String} maskType?
 38  */
 39 AppKit.prototype.showLoadMask = function(app, maskType) {
 40 	var isPopup = false;
 41 	if(app.getHost() && app.getHost().modal === true){
 42 		isPopup = true;
 43 	}
 44 	this.hideLoadMask(app);
 45 	
 46 	if(app.isUDCInstance()){
 47 		app = app.getHostAppInstance();
 48 	}
 49 		
 50 	var showConstraint = {
 51 			"position" : "absolute",
 52 			"top" : "0",
 53 			"bottom" : "0",
 54 			"left" : "0",
 55 			"right" : "0"
 56 	};
 57 	showConstraint = this.wrapConstraints(app, showConstraint);
 58 	
 59 	//var container = isPopup ? app.getContainer() : app.getRootAppInstance().getContainer();
 60 	var container = app.getContainer();
 61 	var layout = container.getLayout();
 62 	
 63 	var loadmask = container.getAppInstance().lookup("__loadmask__");
 64 	
 65 	if(maskType == "pro") {
 66 		loadmask = container.getAppInstance().lookup("__loadmask_pro__");
 67 		if(loadmask) {
 68 			container.replaceConstraint(loadmask, showConstraint);
 69 		} else {
 70 			loadmask = new udc.com.loadmaskprogress("__loadmask_pro__");
 71 			if(layout instanceof cpr.controls.layouts.FormLayout){
 72 				container.floatControl(loadmask, showConstraint);
 73 			}else{
 74 				container.addChild(loadmask, showConstraint);
 75 			}
 76 			container.getAppInstance().register(loadmask);
 77 		}
 78 		loadmask.module.start();
 79 	} else {
 80 		loadmask = container.getAppInstance().lookup("__loadmask__");
 81 		try{
 82 			
 83 			if(loadmask) {
 84 				if(layout instanceof cpr.controls.layouts.FormLayout){
 85 					app.floatControl(loadmask, showConstraint);
 86 				}else{
 87 					container.replaceConstraint(loadmask, showConstraint);
 88 				}
 89 			} else {
 90 				loadmask = new udc.com.loadmask("__loadmask__");
 91 				
 92 				if(layout instanceof cpr.controls.layouts.FormLayout){
 93 					container.floatControl(loadmask, showConstraint);
 94 				}else{
 95 					container.addChild(loadmask, showConstraint);
 96 				}
 97 				container.getAppInstance().register(loadmask);
 98 			}
 99 		}catch(ex){showConstraint = null;}
100 	}
101 	
102 	try{
103 		if(loadmask) {
104 			if(layout instanceof cpr.controls.layouts.FormLayout
105 				|| layout instanceof cpr.controls.layouts.VerticalLayout){
106 				app.floatControl(loadmask, showConstraint);
107 			}else{
108 				container.replaceConstraint(loadmask, showConstraint);
109 			}
110 		} else {
111 			
112 			//loadmask = this._loadMaskUdc;
113 			loadmask = new udc.com.loadmask("__loadmask__");
114 			
115 			if(layout instanceof cpr.controls.layouts.FormLayout
116 				|| layout instanceof cpr.controls.layouts.VerticalLayout){
117 				app.floatControl(loadmask, showConstraint);
118 			}else{
119 				container.addChild(loadmask, showConstraint);
120 			}
121 			container.getAppInstance().register(loadmask);
122 		}
123 	}catch(ex){showConstraint = null;}
124 	
125 	this._activeLoadMask = loadmask;
126 };

128 /**
129  * LoadMask를 감춤
130  * @param {cpr.core.AppInstance} app 앱인스턴스
131  */
132 AppKit.prototype.hideLoadMask = function(app) {
133 	
134 	if(this._activeLoadMask) {
135 		if(this._activeLoadMask.module && this._activeLoadMask.module.end) {
136 			this._activeLoadMask.module.end();
137 		}
138 		var hideConstraint = {
139 				"position" : "absolute",
140 				"top" : "-1px",
141 				"left" : "-1px",
142 				"width" : "1px",
143 				"height" : "1px"
144 		};
145 		
146 		//앱 객체가 사라진 경우... ROOT앱을 기본으로 하여 처리
147 		if(app == null || app.getRootAppInstance() == null){
148 			app = this.getMainApp();
149 		}else{
150 			if(app.isUDCInstance()){
151 				app = app.getHostAppInstance();
152 			}
153 		}
154 		
155 		var isPopup = false;
156 		if(app.getHost() && app.getHost().modal === true){
157 			isPopup = true;
158 		}
159 		
160 		//var container = isPopup ? app.getContainer() : app.getRootAppInstance().getContainer();
161 		var container = app.getContainer();
162 		try{
163 			var layout = container.getLayout();
164 			if(layout instanceof cpr.controls.layouts.FormLayout
165 				|| layout instanceof cpr.controls.layouts.VerticalLayout){
166 				app.removeFloatingControl(this._activeLoadMask);
167 			}else{
168 				hideConstraint = this.wrapConstraints(app, hideConstraint);
169 				container.replaceConstraint(this._activeLoadMask, hideConstraint);
170 			}
171 			if(this._activeLoadMask){
172 				this._activeLoadMask.module.count(0);
173 				this._activeLoadMask.module.hide();
174 			}
175 		}catch(ex){hideConstraint = null;}
176 		
177 		this._activeLoadMask = null;
178 	}
179 };

182 /**
183  * App 화면의 Layout에 맞게 컨트롤 배치 조건 래핑
184  * @param {cpr.core.AppInstance} app 앱인스턴스
185  * @param {cpr.controls.layouts.Constraint}constraint 래핑할 배치조건
186  * @param {cpr.core.AppInstance} poApp 앱인스턴스
187  * @returns 래핑된 배치조건
188  */
189 AppKit.prototype.wrapConstraints = function(app, constraint, poApp) {
190 	var isPopup = false;
191 	if(app.getHost() && app.getHost().modal === true){
192 		isPopup = true;
193 	}
194 	
195 	var layout;
196 	var container = null;
197 	if(poApp == null){
198 		//container = isPopup ? app.getContainer() : app.getRootAppInstance().getContainer();
199 		container = app.getContainer();
200 		poApp = isPopup ? app : app.getRootAppInstance();
201 	}else{
202 		container = poApp.getContainer();
203 	}
204 	layout = container.getLayout();
205 	
206 	if (layout instanceof cpr.controls.layouts.ResponsiveXYLayout) {
207 		var positionConstraints = [];
208 		var allMedia = poApp.allSupportedMedias;
209 		allMedia.forEach(function(media) {
210 			var newConst = _.clone(constraint);
211 			newConst["media"] = media;
212 			positionConstraints[positionConstraints.length] = newConst;
213 		});
214 		return {
215 			"positions" : positionConstraints
216 		};
217 	}
218 	
219 	return constraint;
220 };

253 /**
254  * 메인 화면에 데이터 변경사항이 있는지 여부를 체크한다.<br>
255  * 그리드, 폼레이아웃(프리폼) 대상(UDC, EMB 포함)<br>
256  * 그리드 ignoreModify 사용자속성 "Y" 지정시 continue
257  * @param {cpr.core.AppInstance} app 앱인스턴스
258  * @param {String} psAftMsg? 메시지구분
259  * @param {cpr.controls.Container} poContainer? 컨테이너
260  * @returns {Boolean} 데이터 변경여부
261  */
262 AppKit.prototype.isAppModified = function(app, psAftMsg, poContainer) {
263 	var container;
264 
265 	if(poContainer != null){
266 		container = poContainer;
267 	}else{
268 		var mainApp = this.getMainApp(app);
269 		if(mainApp == null) return false;
270 		container = mainApp.getContainer();
271 	}
272 	
273 	var vaDataCtrls = new Array();
274 	
275 	var _this = this;
276 	function getChildRecursive(poContainer){
277 	    var vaChildCtrls = poContainer.getAllRecursiveChildren();
278 	    for (var i=0, len=vaChildCtrls.length; i<len; i++) {
279 	        if (vaChildCtrls[i].type == "grid") {
280 	        	vaDataCtrls.push(vaChildCtrls[i]);
281 	        //입력형식의 폼레이아웃은 form-box CSS로 구분	
282 	        }else if (vaChildCtrls[i] instanceof cpr.controls.Container && vaChildCtrls[i].style.getClasses().indexOf(_this.FreeForm._FORM_LAYOUT_CSS) != -1) {
283 	        	vaDataCtrls.push(vaChildCtrls[i]);
284 	        }else if(vaChildCtrls[i] instanceof cpr.controls.UDCBase){
285 	        	var voUdcApp = vaChildCtrls[i].getEmbeddedAppInstance();
286 	        	if(voUdcApp) getChildRecursive(voUdcApp.getContainer());
287 	        }else if(vaChildCtrls[i] instanceof cpr.controls.EmbeddedApp){
288 	        	var voEmbApp = vaChildCtrls[i].getEmbeddedAppInstance();
289 	        	if(voEmbApp) getChildRecursive(voEmbApp.getContainer());
290 	        }
291 	    }
292 	}
293 	getChildRecursive(container);
294 	
295 	var modify = false;
296 	var ctrl = null;
297 	var vsFieldLabel = "";
298 	for(var i=0, len=vaDataCtrls.length; i<len; i++){
299 		ctrl = vaDataCtrls[i];
300 		if(ctrl.type == "grid"){
301 			
302 			//그리드 수정 내역 체크 무시 attr
303 			if(ctrl.userAttr("ignoreModify") === "Y" || ctrl.dataSet == null) continue;
304 			if(ctrl.dataSet.isModified()){
305 				modify = true;
306 				vsFieldLabel = ctrl.fieldLabel;
307 				break;
308 			}
309 		}else{
310 			var dataSet = this.Group.getBindDataSet(ctrl.getAppInstance(), ctrl);
311 			if(dataSet != null && dataSet.isModified()) {
312 				modify = true;
313 				vsFieldLabel = ctrl.fieldLabel;
314 				break;
315 			}
316 		}
317 	}
318 	//변경사항이 반영되지 않았습니다. 계속 하시겠습니까? confirm
319 	if(modify && psAftMsg != null && psAftMsg.toUpperCase() == "CRM"){
320 		if(!this.Msg.confirm("CRM-M003", [vsFieldLabel])) return true;
321 		else return false;
322 	}
323 	return modify;
324 };

432 /**
433  * 컨트롤(그룹) 또는 Grid의 내의 입력 값에 대한 유효성 체크를 수행한다.
434  * @param {cpr.core.AppInstance} app 앱인스턴스
435  * @param {#uicontrol | Array} paCtlId 컨트롤 ID
436  * @param {"all" | "modify" | "current"}  dataScope? (all:그리드의 전체 데이터, modify:변경된 전체 Row, current:현재  Row)
437  * @returns {Boolean} Valid true, Invalid false. <br>
438  *  	참고 : 1. 그리드에 연결된 데이터셋의 info는 PK컬럼으로 인식<br>
439  *              그리드 초기화시 (Grid.init) info에 설정된 PK컬럼은 필수값으로 지정되고 해당 컬럼에 ignorePk="Y" 사용자속성 부여시 필수여부 체크를 무시한다.<br>
440  *           2. 그리드 + 폼레이아웃(입력폼) 구성이고 그리드 사용자 속성에 bindDataFormId(폼레이아웃ID) 지정시 유효성체크 부적합 셀은  bindDataFormId에 지정된 폼레이아웃의 컨트롤에 포커싱이 간다. 
441  *           
442  */
443 AppKit.prototype.validate = function(app, paCtlId, dataScope) {
444 	
445 	dataScope = dataScope != null ? dataScope : "upd";
446 	if(!(paCtlId instanceof Array)){
447 		paCtlId = [paCtlId];
448 	}
449 	
450 	var valid = true;
451 	for(var i=0, len=paCtlId.length; i<len; i++) {
452 		var ctrlId = paCtlId[i];
453 		var ctrl = app.lookup(paCtlId[i]);
454 		if(ctrl instanceof cpr.controls.Grid){
455 			valid = this._validateGrid(ctrl, dataScope);
456 		}else if(ctrl instanceof cpr.controls.Container){
457 			/** @type cpr.bind.BindContext */
458 			var bindContext = this.Group.getBindContext(app, ctrl);
459 			if(bindContext){
460 				/**@type cpr.data.DataSet */
461 				var dataset = bindContext.grid ? bindContext.grid.dataSet : bindContext.dataSet;
462 				var rowIndex = bindContext.grid ? bindContext.grid.getSelectedRowIndex() : bindContext.rowIndex;
463 				//프리폼의 상태가 삭제상태이면... 유효성 체크에서 제외함
464 				if(dataset.getRowState(rowIndex) == cpr.data.tabledata.RowState.DELETED) continue;
465 				
466 				if(bindContext.grid){
467 					valid = this._validateFreeForm(ctrl, dataset);
468 				}else{
469 					valid = this._validateControl(ctrl);
470 				}
471 			}else{
472 				valid = this._validateControl(ctrl);
473 			}
474 		}else{
475 			valid = this._validateControl(ctrl);
476 		}
477 		
478 		if(valid == false) {
479 			return false;
480 		}
481 	}
482 	
483 	return true;
484 };

486 /**
487  * 일반 컨트롤에 대한 Validation 체크
488  * @param {cpr.controls.UIControl} ctrl
489  * @param {cpr.controls.UIControl} poParentCtl
490  * @private
491  */
492 AppKit.prototype._validateControl = function(ctrl, poParentCtl) {
493 	if(!ctrl) return true;
494 	
495 	var valid = true;
496 	var _this = this;
497 	if(ctrl instanceof cpr.controls.Container) { // Group 일 경우 체크
498 		var children = this._getChildren(ctrl);
499 		var child;
500 		for(var i=0, len=children.length; i<len; i++){
501 			child = children[i];
502 			// 컨트롤별 Validation Check
503 			if(this._validateControl(child, ctrl) == false) {
504 				valid = false;
505 				break;
506 			}
507 		}
508 		return valid;
509 	} else if(ctrl instanceof cpr.controls.UDCBase){ //UDC인 경우
510 		var embApp = ctrl.getEmbeddedAppInstance();
511 		var children = embApp.getContainer().getAllRecursiveChildren();
512 		var child;
513 		for(var i=0, len=children.length; i<len; i++){
514 			child = children[i];
515 			// 컨트롤별 Validation Check
516 			if(this._validateControl(child, ctrl) == false) {
517 				valid = false;
518 				break;
519 			}
520 		}
521 		return valid;
522 	} else {
523 		valid = this.Validator.validate(ctrl, ctrl.value, poParentCtl);
524 		if(valid == false) {

754 /**

757  * @private
758  */
759 AppKit.prototype._getChildren = function(pcGroup) {
760 	var children = pcGroup.getAllRecursiveChildren();
761 	function getNextControls(each,children){
762 		var order = [each];
763 		var next = each;
764 		while(next != null){
765 			next = next.getNextControl();
766 			if(next != null && children.indexOf(next) > -1 && order.indexOf(next) == -1) order.push(next);
767 			else next = null;
768 		}
769 		return order;
770 	} 
771 	
772 	var orderCtrls = [];
773 	children.forEach(function(each){
774 		if(children.indexOf(each.getPrevControl()) ==-1 && each.getNextControl() != null){
775 			orderCtrls = getNextControls(each,children);
776 		}
777 	});
778 	
779 	var etcCtrls = [];
780 	children.forEach(function(each){
781 		if(orderCtrls.indexOf(each) == -1){
782 			etcCtrls.push(each);
783 		}
784 	});
785 	
786 	return orderCtrls.concat(etcCtrls);
