172      * [char] : 문자의 길이.<br/>
173  	 * [utf8] : utf8 기준의 문자 byte size.<br/>
174  	 * [ascii] : ascii 기준의 문자 byte size.
175 	 * @return {Number} 문자열 길이
176      */
177     getLength : function(value, unit) {
178     	if(!unit) unit = "char";
179     	
180 		var length = 0;
181 		switch(unit) {
182 			case "utf8":{
183 //				for(var i = 0, c; c = value.charAt(i++); length += (c >> 11 ? 3 : c >> 7 ? 2 : 1));
184 				for(var i=0, len=value.length; i<len; i++) {
185 				    if(escape(value.charAt(i)).length >= 4)
186 				        length += 3;
187 				    else if(escape(value.charAt(i)) == "%A7")
188 				        length += 3;
189 				    else if(escape(value.charAt(i)) != "%0D")
190 				        length++;
191 				    else length++;
192 				}
193 				break;
194 			}
195 			case "ascii":{
196 				for(var i = 0, c; c = value.charAt(i++); length += c >> 7 ? 2 : 1);
197 				break;
198 			}
199 			default : {
200 				length = value.length;
201 			}
202 		}
203 		
204 		return length;

329      * 날짜 문자열을 Date형으로 변환하여 반환한다.
330      * <pre><code>
331      * DateUtil.toDate("2007-02-09","YYYY-MM-DD");
332  	 * </code></pre>
333 	 * @param {Date} psDateTime			날짜
334 	 * @param {String} psPattern	포맷 문자열(ex: YYYY-MM-DD)
335 	 * @example DateUtil.toDate("2007-02-09","YYYY-MM-DD")
336 	 * @return {Date} 날짜(Date) 객체
337      */ 
338     toDate : function (psDateTime, psPattern) {
339         var vdDate = new Date();
340         var vnIdx, vnCnt;
341 
342         var vsaFmt = ["Y", "M", "D", "H", "m", "s", "S"];
343         var vnFmtLen = vsaFmt.length;
344         var vnPtnLen = psPattern.length;
345         var vnaNums = [vdDate.getFullYear(), vdDate.getMonth()+1, vdDate.getDate(), vdDate.getHours(), vdDate.getMinutes(), vdDate.getSeconds(), vdDate.getMilliseconds()];
346 
347         for (var i = 0; i < vnFmtLen; i++) {
348             vnIdx = psPattern.indexOf(vsaFmt[i]);
349             if (vnIdx != -1) {
350                 vnCnt = 1;
351                 for (var j=vnIdx+1; j < vnPtnLen; j++) {
352                     if (psPattern.charAt(j) != vsaFmt[i]) { break; }
353                     vnCnt++;
354                 }
355                 vnaNums[i] = Number(psDateTime.substring(vnIdx, vnIdx+vnCnt));
356             } else {
357                 if(i==0) vnaNums[0] = 1900;
358                 else if(i==2) vnaNums[2] = 01;
359             }
360         }
361 
362         if (vnaNums[0] < 1900) { // 년도는 검증
363             if (vnaNums[0] <= vdDate.getFullYear() % 100) {
364                 vnaNums[0] += vdDate.getFullYear() - (vdDate.getFullYear() % 100);
365             } else if (vnaNums[0] < 100) {
366                 vnaNums[0] += 1900;
367             } else {
368                 vnaNums[0] = 1900;
369             }
370         }
371 
372         return new Date(vnaNums[0], vnaNums[1]-1, vnaNums[2], vnaNums[3], vnaNums[4], vnaNums[5], vnaNums[6]);
373     },

610 	/**
611 	 * 해당 파일이 업로드 가능한 파일 유형인지 체크한다.
612 	 * @param {Object} poUtil 유틸 클래스 객체 
613 	 * @param {String} psFileNm 파일명
614 	 * @param {String} psLimitFileExt 업로드 가능제한 파일 확장자
615 	 */
616 	checkFileType : function(poUtil, psFileNm, psLimitFileExt){
617 		var vaCheckFileExt = null;
618 		if (!ValueUtil.isNull(psLimitFileExt)) {
619 			psLimitFileExt = psLimitFileExt.replace(/\./ig, "");
620 			vaCheckFileExt = ValueUtil.split(psLimitFileExt, ",");
621 		}
622 		
623 		if(vaCheckFileExt == null || vaCheckFileExt.length < 1){
624 			vaCheckFileExt = FileUtil.getPemitedFileExts();
625 		}
626 		
627 		var isValid = false;
628 		var arrStr = psFileNm.split(".");
629 		var extStr = arrStr [arrStr.length - 1].toUpperCase();
630 		for(var j=0, jlen=vaCheckFileExt.length; j<jlen; j++){
631 			if (extStr == vaCheckFileExt[j].toUpperCase()) {
632 				isValid = true;
633 				break;
634 			}
635 		}
636 		if(!isValid){
637 			//{0} 유형의 파일은 업로드 불가합니다.
638 			poUtil.Msg.alert("WRN-M024", [extStr]);
639 			return isValid;
640 		}
641 		
642 		var vaLimitedFileExt = FileUtil.getLimitedFileExts();
643 		for(var i=0, len=vaLimitedFileExt.length; i<len; i++) {
644 			if (extStr == vaLimitedFileExt[i]) {
645 				//{0} 유형의 파일은 업로드 불가합니다.
646 				poUtil.Msg.alert("WRN-M024", [extStr]);
647 				isValid = false;
648 				break;
649 			}
650 		}
651 		
652 		return isValid;
653 	}
