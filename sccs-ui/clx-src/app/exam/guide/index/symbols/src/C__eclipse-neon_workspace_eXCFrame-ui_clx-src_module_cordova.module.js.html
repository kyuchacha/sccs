 43 /**
 44  * @param {cpr.core.AppInstance} current
 45  * @param {#app} appId
 46  * @param {"top" | "right" | "bottom" | "left" } direction?
 47  * @return {Promise<cpr.core.AppInstance>}
 48  */
 49 globals.pushNewApp = function(app, appId, direction) {
 50 	direction = direction || "right";
 51 
 52 	var appearingTransform;
 53 	var disappearingTransform;
 54 
 55 	switch (direction) {
 56 		case "right":
 57 			appearingTransform = "translate3d(100%, 0px, 0px)";
 58 			disappearingTransform = "translate3d(-100%, 0px, 0px)"
 59 			break;
 60 
 61 		case "left":
 62 			appearingTransform = "translate3d(-100%, 0px, 0px)";
 63 			disappearingTransform = "translate3d(100%, 0px, 0px)"
 64 			break;
 65 
 66 		case "bottom":
 67 			appearingTransform = "translate3d(0px, 100%, 0px)";
 68 			disappearingTransform = "translate3d(0px, -100%, 0px)"
 69 			break;
 70 
 71 		case "top":
 72 			appearingTransform = "translate3d(0px, -100%, 0px)";
 73 			disappearingTransform = "translate3d(0px, 100%, 0px)"
 74 			break;
 75 	}
 76 
 77 	return new Promise(function(resolve, reject) {
 78 		cpr.core.App.load(appId, function(loadedApp) {
 79 			if (!loadedApp) {
 80 				return reject("앱을 로드하지 못했습니다: " + appId);
 81 			}
 82 
 83 			var newAppInstance = loadedApp.createNewInstance();
 84 			newAppInstance.run(null, function() {
 85 				resolve(newAppInstance);
 86 				newAppInstance.getContainer().style.animateFrom({
 87 					"transform": appearingTransform
 88 				});
 89 				app.getContainer().addEventListenerOnce("transitionend", function() {
 90 					app.dispose();
 91 				});
 92 				app.getContainer().style.animateTo({
 93 					"transform": disappearingTransform
 94 				});
 95 			});
 96 		});
 97 	});
 98 }

128 /**
129  * 브라우저를 통해 사진파일 선택
130  * @param {(String)=>void} onSuccess
131  * @param {(any)=>void} onError
132  */
133 function takePhotoFromBrowser(onSuccess, onError) {
134 	// 파일 선택을 위해 인풋 엘리먼트 만듬.
135 	var input = document.createElement("input");
136 	input.type = "file";
137 	input.style.display = "none";
138 	document.body.appendChild(input);
139 
140 	// 사용자가 파일을 선택한 경우에 대한 리스너.
141 	input.addEventListener("change", function(e) {
142 		var selectedFile = input.files[0];
143 
144 		// 파일이 선택된 경우.
145 		if (selectedFile) {
146 			// 선택된 파일을 dataURL 형식으로 읽어 들임.
147 			var fileReader = new FileReader();
148 			fileReader.addEventListener("load", function() {
149 				var dataURL = fileReader.result;
150 				onSuccess(dataURL);
151 			});
152 			fileReader.readAsDataURL(selectedFile);
153 		}
154 		document.body.removeChild(input);
155 	});
156 	input.click();
157 }

164 function takePhotoFromCordova(source, onSuccess, onError) {
165 	if (!navigator.camera) {
166 		return onError("카메라 플러그인을 찾을 수 없습니다.");
167 	}
168 	var sourceTarget = null;
169 	if(source == "camera"){
170 		sourceTarget = Camera.PictureSourceType.CAMERA
171 	}else if(source == "photoLibrary"){
172 		sourceTarget = Camera.PictureSourceType.PHOTOLIBRARY
173 	}
174 	navigator.camera.getPicture(onSuccess, onError, {
175 		targetWidth: 1024,
176 		targetHeight: 1024,
177 		qualitiy: 50,
178 		destinationType: Camera.DestinationType.FILE_URI,
179 		sourceType: sourceTarget,
180 		encodingType: Camera.EncodingType.JPEG,
181 		correctOrientation: true
182 	});
183 }

248 function recordAudio(onSuccess, onError) {
249 	
250 	if (!Media) {
251 		return onError("미디어 플러그인을 찾을 수 없습니다.");
252 	}
253 	
254     var src = "myrecording.mp3";
255     mediaRec = new Media(src, onSuccess, onError);
256 
257     // Record audio
258     mediaRec.startRecord();
259 
260     // Stop recording after 10 sec
261     var recTime = 0;
262     
263    	if(recInterval != null){
264    		return ;
265    	}
266     
267     recInterval = setInterval(function() {
268         recTime = recTime + 1;
269 //        setAudioPosition(recTime + " sec");
270         if (recTime >= 10) {
271             clearInterval(recInterval);
272             recInterval = null;
273             mediaRec.stopRecord();
274         }
275     }, 1000);
276 }
