 36 /**
 37  * 화면에 LoadMask 출력
 38  * 비동기 서브미션 호출시 화면에 로딩 이미지 출력
 39  * @param {cpr.core.AppInstance} app 앱인스턴스
 40  * @param {String} maskType?
 41  */
 42 AppKit.prototype.showLoadMask = function(app, maskType) {
 43 	var isPopup = false;
 44 	if(app.getHost() && app.getHost().modal === true){
 45 		isPopup = true;
 46 	}
 47 	this.hideLoadMask(app);
 48 	
 49 	var showConstraint = {
 50 			"position" : "absolute",
 51 			"top" : "0",
 52 			"bottom" : "0",
 53 			"left" : "0",
 54 			"right" : "0"
 55 	};
 56 	showConstraint = this.wrapConstraints(app, showConstraint);
 57 	
 58 	//var container = isPopup ? app.getContainer() : app.getRootAppInstance().getContainer();
 59 	var container = app.getContainer();
 60 	var layout = container.getLayout();
 61 	
 62 	var loadmask = container.getAppInstance().lookup(this._loadMaskId);
 63 	
 64 	if(maskType == "pro") {
 65 		loadmask = container.getAppInstance().lookup("__loadmask_pro__");
 66 		if(loadmask) {
 67 			container.replaceConstraint(loadmask, showConstraint);
 68 		} else {
 69 			loadmask = new udc.com.loadmaskprogress("__loadmask_pro__");
 70 			if(layout instanceof cpr.controls.layouts.FormLayout){
 71 				container.floatControl(loadmask, showConstraint);
 72 			}else{
 73 				container.addChild(loadmask, showConstraint);
 74 			}
 75 			container.getAppInstance().register(loadmask);
 76 		}
 77 		loadmask.module.start();
 78 	} else {
 79 		loadmask = container.getAppInstance().lookup("__loadmask__");
 80 		try{
 81 			
 82 			if(loadmask) {
 83 				if(layout instanceof cpr.controls.layouts.FormLayout){
 84 					app.floatControl(loadmask, showConstraint);
 85 				}else{
 86 					container.replaceConstraint(loadmask, showConstraint);
 87 				}
 88 			} else {
 89 				loadmask = new udc.com.loadmask("__loadmask__");
 90 				
 91 				if(layout instanceof cpr.controls.layouts.FormLayout){
 92 					container.floatControl(loadmask, showConstraint);
 93 				}else{
 94 					container.addChild(loadmask, showConstraint);
 95 				}
 96 				container.getAppInstance().register(loadmask);
 97 			}
 98 		}catch(ex){showConstraint = null;}
 99 	}
100 	
101 	try{
102 		if(loadmask) {
103 			if(layout instanceof cpr.controls.layouts.FormLayout
104 				|| layout instanceof cpr.controls.layouts.VerticalLayout){
105 				app.floatControl(loadmask, showConstraint);
106 			}else{
107 				container.replaceConstraint(loadmask, showConstraint);
108 			}
109 		} else {
110 			
111 			//loadmask = this._loadMaskUdc;
112 			loadmask = new udc.com.loadmask(this._loadMaskId);
113 			
114 			if(layout instanceof cpr.controls.layouts.FormLayout
115 				|| layout instanceof cpr.controls.layouts.VerticalLayout){
116 				app.floatControl(loadmask, showConstraint);
117 			}else{
118 				container.addChild(loadmask, showConstraint);
119 			}
120 			container.getAppInstance().register(loadmask);
121 		}
122 	}catch(ex){showConstraint = null;}
123 	
124 	this._activeLoadMask = loadmask;
125 };

127 /**
128  * LoadMask를 감춤
129  * @param {cpr.core.AppInstance} app 앱인스턴스
130  */
131 AppKit.prototype.hideLoadMask = function(app) {
132 	if(this._activeLoadMask) {
133 		if(this._activeLoadMask.module && this._activeLoadMask.module.end) {
134 			this._activeLoadMask.module.end();
135 		}
136 		var hideConstraint = {
137 				"position" : "absolute",
138 				"top" : "-1px",
139 				"left" : "-1px",
140 				"width" : "1px",
141 				"height" : "1px"
142 		};
143 		
144 		//앱 객체가 사라진 경우... ROOT앱을 기본으로 하여 처리
145 		if(app == null || app.getRootAppInstance() == null){
146 			app = this.getMainApp();
147 		}
148 		
149 		var isPopup = false;
150 		if(app.getHost() && app.getHost().modal === true){
151 			isPopup = true;
152 		}
153 		
154 		//var container = isPopup ? app.getContainer() : app.getRootAppInstance().getContainer();
155 		var container = app.getContainer();
156 		try{
157 			var layout = container.getLayout();
158 			if(layout instanceof cpr.controls.layouts.FormLayout
159 				|| layout instanceof cpr.controls.layouts.VerticalLayout){
160 				app.removeFloatingControl(this._activeLoadMask);
161 			}else{
162 				hideConstraint = this.wrapConstraints(app, hideConstraint);
163 				container.replaceConstraint(this._activeLoadMask, hideConstraint);
164 			}
165 			if(this._activeLoadMask){
166 				this._activeLoadMask.module.count(0);
167 				this._activeLoadMask.module.hide();
168 			}
169 		}catch(ex){hideConstraint = null;}
170 		
171 		this._activeLoadMask = null;
172 	}
173 };

176 /**
177  * App 화면의 Layout에 맞게 컨트롤 배치 조건 래핑
178  * @param {cpr.core.AppInstance} app 앱인스턴스
179  * @param {cpr.controls.layouts.Constraint}constraint 래핑할 배치조건
180  * @param {cpr.core.AppInstance} poApp 앱인스턴스
181  * @returns 래핑된 배치조건
182  */
183 AppKit.prototype.wrapConstraints = function(app, constraint, poApp) {
184 	var isPopup = false;
185 	if(app.getHost() && app.getHost().modal === true){
186 		isPopup = true;
187 	}
188 	
189 	var layout;
190 	var container = null;
191 	if(poApp == null){
192 		//container = isPopup ? app.getContainer() : app.getRootAppInstance().getContainer();
193 		container = app.getContainer();
194 		poApp = isPopup ? app : app.getRootAppInstance();
195 	}else{
196 		container = poApp.getContainer();
197 	}
198 	layout = container.getLayout();
199 	
200 	if (layout instanceof cpr.controls.layouts.ResponsiveXYLayout) {
201 		var positionConstraints = [];
202 		var allMedia = poApp.allSupportedMedias;
203 		allMedia.forEach(function(media) {
204 			var newConst = _.clone(constraint);
205 			newConst["media"] = media;
206 			positionConstraints[positionConstraints.length] = newConst;
207 		});
208 		return {
209 			"positions" : positionConstraints
210 		};
211 	}
212 	
213 	return constraint;
214 };

247 /**
248  * 메인 화면에 데이터 변경사항이 있는지 여부를 체크한다.<br>
249  * 그리드, 폼레이아웃(프리폼) 대상(UDC, EMB 포함)<br>
250  * 그리드 ignoreModify 사용자속성 "Y" 지정시 continue
251  * @param {cpr.core.AppInstance} app 앱인스턴스
252  * @param {String} psAftMsg? 메시지구분
253  * @returns {Boolean} 데이터 변경여부
254  */
255 AppKit.prototype.isAppModified = function(app, psAftMsg) {
256 	var mainApp = this.getMainApp(app);
257 	if(mainApp == null) return false;
258 	
259 	var vaDataCtrls = new Array();
260 	var container = mainApp.getContainer();
261 	
262 	var _this = this;
263 	
264 	function getChildRecursive(poContainer){
265 	    var vaChildCtrls = poContainer.getAllRecursiveChildren();
266 	    for (var i=0, len=vaChildCtrls.length; i<len; i++) {
267 	        if (vaChildCtrls[i].type == "grid") {
268 	        	vaDataCtrls.push(vaChildCtrls[i]);
269 	        //입력형식의 폼레이아웃은 form-box CSS로 구분	
270 	        }else if (vaChildCtrls[i] instanceof cpr.controls.Container && vaChildCtrls[i].style.getClasses().indexOf(_this.FreeForm._FORM_LAYOUT_CSS) != -1) {
271 	        	vaDataCtrls.push(vaChildCtrls[i]);
272 	        }else if(vaChildCtrls[i] instanceof cpr.controls.UDCBase){
273 	        	var voUdcApp = vaChildCtrls[i].getEmbeddedAppInstance();
274 	        	if(voUdcApp) getChildRecursive(voUdcApp.getContainer());
275 	        }else if(vaChildCtrls[i] instanceof cpr.controls.EmbeddedApp){
276 	        	var voEmbApp = vaChildCtrls[i].getEmbeddedAppInstance();
277 	        	if(voEmbApp) getChildRecursive(voEmbApp.getContainer());
278 	        }
279 	    }
280 	}
281 	getChildRecursive(container);
282 	
283 	var modify = false;
284 	var ctrl = null;
285 	var vsFieldLabel = "";
286 	for(var i=0, len=vaDataCtrls.length; i<len; i++){
287 		ctrl = vaDataCtrls[i];
288 		if(ctrl.type == "grid"){
289 			
290 			//그리드 수정 내역 체크 무시 attr
291 			if(ctrl.userAttr("ignoreModify") === "Y" || ctrl.dataSet == null) continue;
292 			if(ctrl.dataSet.isModified()){
293 				modify = true;
294 				vsFieldLabel = ctrl.fieldLabel;
295 				break;
296 			}
297 		}else{
298 			var dataSet = this.Group.getBindDataSet(ctrl.getAppInstance(), ctrl);
299 			if(dataSet != null && dataSet.isModified()) {
300 				modify = true;
301 				vsFieldLabel = ctrl.fieldLabel;
302 				break;
303 			}
304 		}
305 	}
306 	//변경사항이 반영되지 않았습니다. 계속 하시겠습니까? confirm
307 	if(modify && psAftMsg != null && psAftMsg.toUpperCase() == "CRM"){
308 		if(!this.Msg.confirm("CRM-M003", [vsFieldLabel])) return true;
309 		else return false;
310 	}
311 	return modify;
312 };

314 /**
315  * 메인 화면에 데이터 변경사항이 있는지 여부를 체크한다.

332 	        	vaDataCtrls.push(vaChildCtrls[i]);
333 	        }else if(vaChildCtrls[i] instanceof cpr.controls.UDCBase){
334 	        	var voUdcApp = vaChildCtrls[i].getEmbeddedAppInstance();
335 	        	if(voUdcApp) getChildRecursive(voUdcApp.getContainer());
336 	        }else if(vaChildCtrls[i] instanceof cpr.controls.EmbeddedApp){
337 	        	var voEmbApp = vaChildCtrls[i].getEmbeddedAppInstance();
338 	        	if(voEmbApp) getChildRecursive(voEmbApp.getContainer());
339 	        }
340 	    }
341 	}

343 	
344 	var ctrl = null;
345 	for(var i=0, len=vaDataCtrls.length; i<len; i++){
346 		ctrl = vaDataCtrls[i];

412 /**
413  * 컨트롤(그룹) 또는 Grid의 내의 입력 값에 대한 유효성 체크를 수행한다.
414  * @param {cpr.core.AppInstance} app 앱인스턴스
415  * @param {#uicontrol | Array} paCtlId 컨트롤 ID
416  * @param {"all" | "modify" | "current"}  dataScope? (all:그리드의 전체 데이터, modify:변경된 전체 Row, current:현재  Row)
417  * @returns {Boolean} Valid true, Invalid false. <br>
418  *  	참고 : 1. 그리드에 연결된 데이터셋의 info는 PK컬럼으로 인식<br>
419  *              그리드 초기화시 (Grid.init) info에 설정된 PK컬럼은 필수값으로 지정되고 해당 컬럼에 ignorePk="Y" 사용자속성 부여시 필수여부 체크를 무시한다.<br>
420  *           2. 그리드 + 폼레이아웃(입력폼) 구성이고 그리드 사용자 속성에 bindDataFormId(폼레이아웃ID) 지정시 유효성체크 부적합 셀은  bindDataFormId에 지정된 폼레이아웃의 컨트롤에 포커싱이 간다. 
421  *           
422  */
423 AppKit.prototype.validate = function(app, paCtlId, dataScope) {
424 	
425 	dataScope = dataScope != null ? dataScope : "upd";
426 	if(!(paCtlId instanceof Array)){
427 		paCtlId = [paCtlId];
428 	}
429 	
430 	var valid = true;
431 	for(var i=0, len=paCtlId.length; i<len; i++) {
432 		var ctrlId = paCtlId[i];
433 		var ctrl = app.lookup(paCtlId[i]);
434 		if(ctrl instanceof cpr.controls.Grid){
435 			valid = this._validateGrid(ctrl, dataScope);
436 		}else if(ctrl instanceof cpr.controls.Container){
437 			/** @type cpr.bind.BindContext */
438 			var voBindContext = this.Group.getBindContext(app, ctrl);
439 			if(voBindContext){
440 				/**@type cpr.data.DataSet */
441 				var voDs = voBindContext.grid ? voBindContext.grid.dataSet : voBindContext.dataSet;
442 				var rowIndex = voBindContext.grid ? voBindContext.grid.getSelectedRowIndex() : voBindContext.rowIndex;
443 				//프리폼의 상태가 삭제상태이면... 유효성 체크에서 제외함
444 				if(voDs.getRowState(rowIndex) == cpr.data.tabledata.RowState.DELETED) continue;
445 			}
446 			
447 			valid = this._validateControl(ctrl);
448 		}else{
449 			valid = this._validateControl(ctrl);
450 		}
451 		
452 		if(valid == false) {
453 			return false;
454 		}
455 	}
456 	
457 	return true;
458 };

460 /**
461  * 일반 컨트롤에 대한 Validation 체크
462  * @param {cpr.controls.UIControl} ctrl
463  * @param {cpr.controls.UIControl} poParentCtl
464  * @private
465  */
466 AppKit.prototype._validateControl = function(ctrl, poParentCtl) {
467 	if(!ctrl) return true;
468 	
469 	var valid = true;
470 	var _this = this;
471 	if(ctrl instanceof cpr.controls.Container) { // Group 일 경우 체크
472 		var children = this._getChildren(ctrl);
473 		var child;
474 		for(var i=0, len=children.length; i<len; i++){
475 			child = children[i];
476 			// 컨트롤별 Validation Check
477 			if(this._validateControl(child, ctrl) == false) {
478 				valid = false;
479 				break;
480 			}
481 		}
482 		return valid;
483 	} else if(ctrl instanceof cpr.controls.UDCBase){ //UDC인 경우
484 		var embApp = ctrl.getEmbeddedAppInstance();
485 		var children = embApp.getContainer().getAllRecursiveChildren();
486 		var child;
487 		for(var i=0, len=children.length; i<len; i++){
488 			child = children[i];
489 			// 컨트롤별 Validation Check
490 			if(this._validateControl(child, ctrl) == false) {
491 				valid = false;
492 				break;
493 			}
494 		}
495 		return valid;
496 	} else {
497 		valid = this.Validator.validate(ctrl, ctrl.value, poParentCtl);
498 		if(valid == false) {
499 			//탭내에 컨트롤이 존재하는 경우... 해당 탭페이지 포커싱
500 			this._focusToTabItem(ctrl);
501 			ctrl.focus();
502 		}
503 		return valid;
504 	}
505 };

507 /**
508  * Grid의 변경된 전체 데이터에 대한 Validation 체크<br>
509  * - 사이트별 Customizing 필요<br>
510  * 가능한 한 Validation 체크시 validate 메소드를 사용
511  * @param {cpr.controls.Grid} poGrid 체크할 Grid
512  * @param {"all" | "modify" | "current"} dataScope all:그리드의 전체 데이터, modify:변경된 전체 Row, current:현재  Row
513  * @returns {Boolean}
514  * @private
515  */
516 AppKit.prototype._validateGrid = function(poGrid, dataScope) {
517 	dataScope = dataScope != null ? dataScope : "modify";
518 	/** @type cpr.controls.Grid */
519 	var grd = poGrid;
520 	if(!grd) return false;
521 	
522 	var vsDataBindCtxId = grd.userAttr("bindDataFormId");
523 	
524 	var _this = this;
525 	/**
526 	 * @type cpr.controls.gridpart.GridBand
527 	 */
528 	var detailBand = grd.detail;
529 	var cellCnt = detailBand.cellCount;
530 	
531 	/**
532 	 * @type cpr.data.DataSet
533 	 */
534 	var dataSet = grd.dataSet;
535 	var rowIndexs = null;
536 	if(dataScope == "all"){
537 		rowIndexs = dataSet.getRowStatedIndices(cpr.data.tabledata.RowState.INSERTED | cpr.data.tabledata.RowState.UPDATED | cpr.data.tabledata.RowState.DELETED | cpr.data.tabledata.RowState.UNCHANGED);
538 	}else{
539 		rowIndexs = dataSet.getRowStatedIndices(cpr.data.tabledata.RowState.INSERTED | cpr.data.tabledata.RowState.UPDATED);
540 	}
541 	var _this = this;
542 	var invalid = rowIndexs.some(function(idx) {
543 		var row = dataSet.getRow(idx);
544 		var col = null;
545 		for(var i = 0; i < cellCnt; i++) {
546 			/**  @type cpr.controls.gridpart.GridColumn */
547 			col = detailBand.getColumn(i);
548 			//컬럼 매핑노드가 없으면... SKIP
549 			if(col.columnName == null || col.columnName == "") continue;
550 			if(col.columnType == "checkbox" || col.columnType == "rowindex") continue;
551 			//컬럼 유형이 output이면... SKIP
552 			if(col.controlType == null || col.controlType == "output" || col.controlType == "button" || col.controlType == "img") continue;
553 			//신규행  PK 체크 무시... SKIP
554 			if(row.getState() == cpr.data.tabledata.RowState.INSERTED && (col.control && col.control.userAttr("ignorePk") == "Y")) continue;
555 			
556 			// 컨트롤별 Validation Check
557 			if(_this.Validator.validate(col.control, row.getValue(col.columnName), grd, idx, i) == false) {
558 				//유효성 체크로 인해 selection-change 발생여부 셋팅 
559 				grd.userAttr("selectionChangeByValidation", "true");
560 				//탭내에 컨트롤이 존재하는 경우... 해당 탭페이지 포커싱
561 				_this._focusToTabItem(grd);
562 				if(ValueUtil.isNull(vsDataBindCtxId)){
563 					grd.setEditRowIndex(idx, true);
564 					grd.focusCell(idx, i);
565 					//포커싱할 컬럼이 UDC인 경우에...
566 					var dctrl = grd.detail.getColumn(i).control;
567 					if(dctrl instanceof cpr.controls.UDCBase){
568 						var empApp = dctrl.getEmbeddedAppInstance();
569 						dctrl = AppUtil.getUDCBindValueControl(dctrl);
570 						if(dctrl) empApp.focus(dctrl.id);
571 					}
572 				}else{
573 					grd.selectRows(idx);
574 					var cctrl = _this.Group.getDataBindedControl(dataSet.getAppInstance(), vsDataBindCtxId, col.columnName);
575 					if(cctrl) _this.Control.setFocus(cctrl.getAppInstance(), cctrl.id);
576 				}
577 				
578 				return true;
579 			}
580 		}
581 		return false;
582 	});
583 	if(invalid == true) {
584 		return false;
585 	}
586 	
587 	return true;
588 };

591 /**
592  * Validation 체크시 컨트롤이 속한 탭폴더 선택용
593  * @param {cpr.controls.UIControl} ctrl 컨트롤 객체
594  * @private
595  */
596 AppKit.prototype._focusToTabItem = function(ctrl) {
597 	/**@type cpr.controls.TabFolder */
598 	var tab = null;
599 	ctrl.findParent(function(pctrl){
600 		if(pctrl instanceof cpr.controls.TabFolder){
601 			tab = pctrl;
602 			return true;
603 		}
604 		return false;
605 	});
606 	if(tab){
607 		var tabItem = null;
608 		var tabItems = tab.getTabItems();
609 		ctrl.findParent(function(pctrl){
610 			tabItems.some(function(each){
611 				if( each.content == pctrl){
612 					tabItem = each;
613 					return true;
614 				}
615 				return false;
616 			});
617 			return tabItem != null;
618 		});
619 		if(tabItem && tabItem != tab.getSelectedTabItem()){
620 			tab.setSelectedTabItem(tabItem);
621 		}
622 	}
623 };

625 /**
626  * 그룹 컨트롤내의 자식 컨트롤 목록을 반환한다.
627  * @param {cpr.controls.Container} pcGroup 그룹컨트롤
628  * @private
629  */
630 AppKit.prototype._getChildren = function(pcGroup) {
631 	var children = pcGroup.getAllRecursiveChildren();
632 	function getNextControls(each,children){
633 		var order = [each];
634 		var next = each;
635 		while(next != null){
636 			next = next.getNextControl();
637 			if(next != null && children.indexOf(next) > -1 && order.indexOf(next) == -1) order.push(next);
638 			else next = null;
639 		}
640 		return order;
641 	} 
642 	
643 	var orderCtrls = [];
644 	children.forEach(function(each){
645 		if(children.indexOf(each.getPrevControl()) ==-1 && each.getNextControl() != null){
646 			orderCtrls = getNextControls(each,children);
647 		}
648 	});
649 	
650 	var etcCtrls = [];
651 	children.forEach(function(each){
652 		if(orderCtrls.indexOf(each) == -1){
653 			etcCtrls.push(each);
654 		}
655 	});
656 	
657 	return orderCtrls.concat(etcCtrls);
658 };

709 	var vsData = null;
710 	if(_mainApp.__menuInfo != null){

714 		/** @type cpr.controls.MDIFolder */

722 		}
