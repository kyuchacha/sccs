172      * [char] : 문자의 길이.<br/>

174  	 * [ascii] : ascii 기준의 문자 byte size.

177     getLength : function(value, unit) {
178     	if(!unit) unit = "char";
179     	
180 		var length = 0;
181 		switch(unit) {
182 			case "utf8":{
183 //				for(var i = 0, c; c = value.charAt(i++); length += (c >> 11 ? 3 : c >> 7 ? 2 : 1));
184 				for(var i=0, len=value.length; i<len; i++) {

186 				        length += 3;
187 				    else if(escape(value.charAt(i)) == "%A7")
188 				        length += 3;
189 				    else if(escape(value.charAt(i)) != "%0D")
190 				        length++;
191 				    else length++;
192 				}
193 				break;
194 			}
195 			case "ascii":{
196 				for(var i = 0, c; c = value.charAt(i++); length += c >> 7 ? 2 : 1);
197 				break;
198 			}
199 			default : {
200 				length = value.length;
201 			}
202 		}
203 		
204 		return length;

329      * 날짜 문자열을 Date형으로 변환하여 반환한다.
330      * <pre><code>
331      * DateUtil.toDate("2007-02-09","YYYY-MM-DD");
332  	 * </code></pre>
333 	 * @param {Date} psDateTime			날짜
334 	 * @param {String} psPattern	포맷 문자열(ex: YYYY-MM-DD)
335 	 * @example DateUtil.toDate("2007-02-09","YYYY-MM-DD")
336 	 * @return {Date} 날짜(Date) 객체
337      */ 
338     toDate : function (psDateTime, psPattern) {
339         var vdDate = new Date();
340         var vnIdx, vnCnt;
341 
342         var vsaFmt = ["Y", "M", "D", "H", "m", "s", "S"];
343         var vnFmtLen = vsaFmt.length;
344         var vnPtnLen = psPattern.length;
345         var vnaNums = [vdDate.getFullYear(), vdDate.getMonth()+1, vdDate.getDate(), vdDate.getHours(), vdDate.getMinutes(), vdDate.getSeconds(), vdDate.getMilliseconds()];
346 
347         for (var i = 0; i < vnFmtLen; i++) {
348             vnIdx = psPattern.indexOf(vsaFmt[i]);
349             if (vnIdx != -1) {
350                 vnCnt = 1;
351                 for (var j=vnIdx+1; j < vnPtnLen; j++) {
352                     if (psPattern.charAt(j) != vsaFmt[i]) { break; }
353                     vnCnt++;
354                 }
355                 vnaNums[i] = Number(psDateTime.substring(vnIdx, vnIdx+vnCnt));
356             } else {
357                 if(i==0) vnaNums[0] = 1900;
358                 else if(i==2) vnaNums[2] = 01;
359             }
360         }
361 
362         if (vnaNums[0] < 1900) { // 년도는 검증
363             if (vnaNums[0] <= vdDate.getFullYear() % 100) {
364                 vnaNums[0] += vdDate.getFullYear() - (vdDate.getFullYear() % 100);
365             } else if (vnaNums[0] < 100) {
366                 vnaNums[0] += 1900;
367             } else {
368                 vnaNums[0] = 1900;
369             }
370         }
371 
372         return new Date(vnaNums[0], vnaNums[1]-1, vnaNums[2], vnaNums[3], vnaNums[4], vnaNums[5], vnaNums[6]);
373     },

549 	 * @param {String[]} paFileFilter 가능 파일 확장자 명

551 	checkFileExt : function(psFileNm, paFileFilter) {

553 		
554 		var vbCheck = false;
555 		//허용 파일 확장자 체크

558 		for (var i=0, len=paFileFilter.length; i<len; i++) {
559 			if (extStr == paFileFilter[i].toUpperCase()) {
560 				vbCheck = true;
561 				break;
562 			}
563 		}
564 		
565 		if (!vbCheck) {
566 			alert(this.__getMessage("NLS-ERR-M007", [extStr]));
567 			return false;
568 		}
569 		
570 		//제한 파일 확장자 체크
571 		var vaLimitedFileExts = this.getLimitedFileExts();
572 		for (var i=0, len=vaLimitedFileExts.length; i<len; i++) {
573 			if (extStr == vaLimitedFileExts[i]) {
574 				alert(this.__getMessage("NLS-ERR-M007", [extStr]));
575 				return false;
576 			}
577 		}
578 		
579 		return true;

599 	/**
600 	 * 메시지를 반환한다.
601 	 * @param {String} psMessageId
602 	 * @param {String[]} paArgs
603 	 */
604 	__getMessage : function(psMessageId, paArgs){
605 		var message = cpr.I18N.INSTANCE.message(psMessageId);
606 		var index = 0, count = 0;
607     	while ((index = message.indexOf("@", index)) != -1) {
608 	        if (paArgs[count] == null) paArgs[count] = "";
609 	        message = message.substr(0, index) + String(paArgs[count]) + message.substring(index + 1);
610 	        index = index + String(paArgs[count++]).length;
611 	    }
612 	    
613 	    return message.replace(/\\n/ig, "\n");
614 	},

625    		app.getContainer().getChildren().forEach(function(each){
626 			if(each instanceof cpr.controls.FileInput && each.id == "com_fileinput") {

633 		//파일 확장자

635 			var tempFilter = "";
636 			if(paFileFilter instanceof Array) {
637 				for(var i = 0; i < paFileFilter.length; i++) {
638 					if(i == 0) {
639 						tempFilter += "." + paFileFilter[i];

642 					}
643 				}

645 				fileInput.acceptFile = tempFilter;

662 		
663 		fileInput.addEventListenerOnce("value-change", function(e) {
664 			if(typeof (poCallBackFunc) == "function"){

666 				var vsFileNm = "";
