 11 /*
 12  * 본 모듈은 MDI폴더에 포함되어있는 임베디드 앱을 분할해서 띄울 수 있는 모듈입니다.
 13  * 분할영역은 총 기본(전체화면), 2분할, 4분할, 계단형 입니다.
 14  * 2분할과 4분할은 폼레이아웃에 배치되는 형태이며, resize가 가능합니다.
 15  * 계단형은 다이얼로그 형태로 제공 됩니다.
 16  * 분할되어있는 상태에서는 탭헤더가 보이지 않습니다.
 17  * 
 18  * [주의] 
 19  * 1. MDI에 추가되어 있는 탭아이템은 모두 임베디드앱을 포함해야 합니다.
 20  * 2. MDI폴더가 로드되었을 때, 항상 기본분할로 선택되어 있어야 합니다.
 21  * 
 22  * [사용방법]
 23  * 본 모듈을 사용하기 위해, 프로젝트 내에 배치하신 다음, 스크립트에서 comDivideScr(app)을 선언한 뒤에
 24  * 아래의 [호출 가능한 메서드]를 호출하여 사용할 수 있습니다.
 25  * 
 26  * [호출 가능한 메서드]
 27  *  1. divide : MDI에 추가되어있는 임베디드앱을 분할하는 메서드입니다.
 28  *  2. removeAllFloat : 플로팅되어있는 모든 컨트롤을 삭제합니다.
 29  * 							모듈에서 플로팅한 컨트롤 외에 플로팅 되어있는 컨트롤까지 삭제할 수 있습니다.
 30  *  3. updateFloatConstraint : 메인의 메뉴(left, right)가 열릴 때 분할영역의 위치상태를 변경합니다.
 31  *  4. selectMenu : 메뉴를 선택했을 때, MDI폴더 또는 분할영역에 임베디드앱을 추가합니다.
 32  *  5. closeMenu : 열려있는 메뉴를 닫을 때 사용합니다. 
 33  * 		- 선택닫기 : 선택되어있는 탭을 닫습니다. 단, 분할되어있는 경우에는 선택닫기는 불가합니다.
 34  * 						계단형으로 분할되어있을 때는 각각의 다이얼로그를 직접 닫을 수 있습니다.
 35  * 		- 모두닫기 : [홈] 화면을 제외한 나머지 화면을 닫습니다.
 36  *  6. getDivGroup : 현재 플로팅 되어있는 영역의 그룹을 반환합니다. (단, 2분할 또는 4분할되어 있는 폼레이아웃만 해당됩니다.)
 37  */

223 	
224 	var control = e.control;
225

227 	if(vcFocusEmb instanceof cpr.controls.UDCBase && vcFocusEmb.getAppInstance()) {
228 		vcFocusEmb = vcFocusEmb.getAppInstance().getHost();
229 	}
230 	if(vcFocusEmb) {
231 		if(mcFocusEmb != vcFocusEmb) {
232 			mcFocusEmb = vcFocusEmb;
233 			if(mcFloatGrp || moDialogContent.length > 0) {
234 				cpr.core.NotificationCenter.INSTANCE.post("check-menu-auth", {
235 					vcEmb : vcFocusEmb
236 				});
237 			}
238 		}
239 	}

268 					}
269 				});

273

327 /**
328  * 플로팅 컨트롤 모두 삭제
329  */
330 comDivideScr.prototype.removeAllFloat = function () {
331 	
332 	if(moRootAppIns) {
333 		
334 		/* 고정탭이 있는경우 첫번째 아이템 저장 */
335 		if(mbFixFirst) {
336 			if(mcFloatGrp && mcFloatGrp.getChildrenCount() > 0) {
337 				mcMainTab.getTabItems()[0].content = mcFloatGrp.getChildren()[0];
338 			} else if(moDialogContent.length > 0){
339 				var voItem = new cpr.controls.TabItem();
340 				voItem.content  = moDialogContent.map(function(each){
341 					if(each.app.id == msFixAppId) return each;
342 				})[0];
343 				voItem.text = "Dashboard";
344 				voItem.visible = false;
345 				mcMainTab.addTabItem(voItem);
346 			}
347 		}
348 		
349 		mcMainTab.getAppInstance().dialogManager.closeAll();
350 		
351 		// container에 추가해서 삭제로 변경
352 		if(moRootAppIns.getContainer().getChild("grpFloat")) {
353 			moRootAppIns.getContainer().removeChild(mcFloatGrp,true);
354 		}
355 		
356 		moDialogContent = []; // 다이얼로그로 띄웠던 항목 삭제
357 		mcFloatGrp = null; // 2 또는 4분할 삭제
358 		mcNotifier = null; // 알림컨트롤 삭제
359 		
360 		mcMainTab.getAppInstance().removeFloatingControl(mcGrpBoundary);
361 		mcGrpBoundary = null;
362 		
363 	}
364 		
365 	cpr.core.DeferredUpdateManager.INSTANCE.update();
366 }

369 /**
370  * 화면을 분할합니다.
371  * @param {String} psLayout 분할하고자 하는 구성(기본 : default, 2분할 : vertical, 4분할 : division, 계단형 : window)
372  * @param {#notifier} psNotifier 알림컨트롤 ID (분할영역이 있을 경우 알림컨트롤도 분할영역 상단으로 함께 플로팅됩니다.)
373  * @param {cpr.controls.TabItem[]|cpr.controls.Item[]} poSelection?
374  */
375 comDivideScr.prototype.divide = function (psLayout, psNotifier, poSelection) {
376 	
377 	if(mcMainTab == null) return;
378 	
379 	// 1. 앱의 루트앱인스턴스
380 	moRootAppIns = this._app;
381 	
382 	// 2. 메인 MDI 폴더
383 	var voMdiConst = mcMainTab.getOffsetRect();
384 	var voMdiActulRect = mcMainTab.getActualRect();
385 	var voRootConst = moRootAppIns.getActualRect();
386 	
387 	// TODO 플로팅 위치를 변경하기 위해서 아래의 Constraint를 변경하십시오.
388 	var vnTabHeight = mcMainTab.getHeaderControls()[0].getActualRect().height;
389 	mcMainTab.getHeaderControls().forEach(function(each){
390 		if(each.getActualRect().height > vnTabHeight) vnTabHeight = each.getActualRect().height;
391 	});
392 		
393 	// 3. 알림컨트롤
394 	var vcNotifier = moRootAppIns.lookup(psNotifier);
395 	if(vcNotifier && vcNotifier.isFloated()) {
396 		// 다시 기본화면으로 돌아갈 때, 플로팅 되어있던 알림컨트롤 추가
397 		var vnTabHeight = mcMainTab.getHeaderControls()[0].getActualRect().height;
398 		mcMainTab.getHeaderControls().forEach(function(each){
399 			if(each.getActualRect().height > vnTabHeight) vnTabHeight = each.getActualRect().height;
400 		});
401 		this._app.getContainer().addChild(vcNotifier, {
402 			left : voMdiConst.left + "px",
403 			bottom : voMdiConst.bottom + vnTabHeight + "px",
404 			right : voMdiConst.right,
405 			height : vcNotifier.getActualRect().height + "px"
406 		});
407 	}
408 	
409 	var vnDiv = 1; // 화면분할 개수
410 	switch(psLayout){
411 		case msHorizontal : case msVertical : {
412 			vnDiv = 2;
413 			break;
414 		}
415 		case msDivision : {
416 			vnDiv = 4;
417 			break;
418 		}
419 		case msCascading : {
420 			vnDiv = psLayout;
421 			break;
422 		}
423 	}
424 	
425 	// 레이아웃을 변경할 때마다, 임베디드앱의 변경사항을 메인 탭에 저장한다.
426 	if(moDialogContent.length == 0) {
427 		
428 		/* 2 or 4분할의 앱인스턴스 저장 */
429 		var vcFloatGrp = moRootAppIns.lookup("grpFloat");
430 		
431 		if(vcFloatGrp) {
432 			// 분할 변경 시, 기존의 포틀릿(드래그,드롭 객체) 삭제
433 			removeDragManager(mcFloatGrp);
434 
435 			var voChildren = vcFloatGrp.getChildren();
436 			voChildren.forEach(function(each, idx) {
437 				var vnIndex = null;
438 				mcMainTab.getTabItems().map(function(item, index){
439 					if(each.userAttr("tabId") == item.name) vnIndex = index;
440 					
441 					if(vnIndex != null) {
442 						mcMainTab.getTabItems()[vnIndex].content = each;
443 						return;
444 					}
445 				});
446 			});
447 			
448 			if(mcMainTab.userAttr(ATTR_FIXED_ITEM) === "Y" && mcMainTab.userAttr(ATTR_FIXED_ID) && mcMainTab.userAttr(ATTR_FIXED_ID) !== "") {
449 				mcMainTab.getTabItems()[0].visible = false;
450 			}
451 		}
452 		
453 	} else {
454 		/* 계단형의 앱인스턴스 저장 */
455 		for(var idx = 0; idx < moDialogContent.length; idx++){
456 			var voTabItem = new cpr.controls.TabItem();
457 			voTabItem.content = moDialogContent[idx];
458 			voTabItem.text = moDialogContent[idx].userAttr("appId");
459 			voTabItem.closable = moDialogContent[idx].userAttr("closable")=="true" ? true : false;
460 			mcMainTab.addTabItem(voTabItem);
461 			
462 			if(idx == 0 && mcMainTab.userAttr(ATTR_FIXED_ITEM) === "Y" && mcMainTab.userAttr(ATTR_FIXED_ID) && mcMainTab.userAttr(ATTR_FIXED_ID) !== "") {
463 				voTabItem.visible = false;
464 			}
465 		}
466 	}
467 
468 	var voTabItems = mcMainTab.getTabItems();
469 	voTabItems.forEach(function(each){
470 		// tabItem.name 설정 (content 가 null 일 경우 탭아이템 구분가능)
471 		if(each.name == null || each.name == "") {
472 			each.name = each.content.uuid;
473 		}
474 		
475 		// 포커스 되어있던 아이템 선택
476 		if(each.content == mcFocusEmb) {
477 			mcMainTab.setSelectedTabItem(each);
478 		}
479 	})
480 	
481 	// 선택한 아이템
482 	var voSelectedTabItem = [];
483 	if(poSelection) {
484 		poSelection.forEach(function(each){
485 			if(each instanceof cpr.controls.TabItem) {
486 				voSelectedTabItem.push(each);
487 			} else {
488 				// 동일한 앱아이디 화면이 2개 이상일 경우, 각 content 식별을 itemIndex로 설정
489 				var vnItemIndex = each.row.getValue("value");
490 				var tabItem = null;
491 				if(isNaN(vnItemIndex)) {
492 					// window에서 다른 분할형태로 바뀔 경우, embeddedappId 
493 					tabItem = voTabItems.filter(function(each){
494 						if(each.content.id == vnItemIndex) return each;
495 					})[0];
496 				} else {
497 					tabItem = voTabItems[vnItemIndex];
498 				}
499 				if(tabItem) {
500 					voSelectedTabItem.push(tabItem);
501 				}
502 			}
503 		});
504 	}
505 	
506 	// 플로팅 컨트롤 모두 삭제
507 	this.removeAllFloat();
508 
509 	if(psLayout == msDefault) {
510 		/* 전체화면 */
511 		
512 		// 탭아이템 헤더 visible하도록 설정
513 		voTabItems.forEach(function (each, index) {
514 			if(mcMainTab.userAttr(ATTR_FIXED_ITEM) === "Y" && mcMainTab.userAttr(ATTR_FIXED_ID) && mcMainTab.userAttr(ATTR_FIXED_ID) !== "" 
515 			&& index == 0) {
516 				each.visible = false;
517 			} else {
518 				each.visible = true;
519 			}
520 		});
521 		
522 		if(voSelectedTabItem.length > 0) {

1059 /**
1060  * 선택/모두 닫기
1061  * @param {cpr.controls.ComboBox} pcCombobox
1062  * @param {String} psUnit 선택닫기 : "select" / 모두닫기 : "all" / 고정탭제외 닫기 : "fixClose"
1063  * @param {cpr.controls.TabItem} pcCloseItem? 닫을 탭아이템
1064  */
1065 comDivideScr.prototype.closeMenu = function (pcCombobox, psUnit, pcCloseItem) {
1066 	
1067 	if(mcMainTab == null) return;
1068 	
1069 	if(psUnit == "all") {
1070 		/* 모두닫기 */
1071 		this.removeAllFloat();
1072 		mcMainTab.closeAll();
1073 		
1074 	} else if(psUnit == "fixClose") {
1075 		/* 고정 탭 제외 닫기 */
1076 		var voApp = mcMainTab.getAppInstance();
1077 		var vcFixedId = mcMainTab.userAttr(ATTR_FIXED_ID);
1078 		voApp.lookup(vcFixedId).visible = true;
1079 		
1080 		var voFixedTabItem = mcMainTab.getTabItems()[0];
1081 		mcMainTab.closeOthers(voFixedTabItem);
1082 
1083 		this.removeAllFloat();
1084 	} else {
1085 		/* 선택닫기 */
1086 		if(pcCloseItem) {
1087 			mcMainTab.close(pcCloseItem);	
1088 			mcMainTab.setSelectedTabItem(mcMainTab.getTabItems()[0]);
1089 		} else {
1090 			cpr.core.NotificationCenter.INSTANCE.post("app-msg", {
1091 				msg : "닫을 화면을 설정하세요."
1092 			});
1093 			return;
1094 		}
1095 		
1096 		if(mcFloatGrp) {
1097 			// 2분할 or 4분할 초기화
1098 			this.divide(msDefault, mcNotifier); 
1099 		}
1100 	}
1101 	
1102 	if(pcCombobox) {
1103 		pcCombobox.selectItemByValue("default", true);
1104 	}
1105 }
